class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int totalsum = 0;
        
        int n =cardPoints.size();
        for(int i =0;i<k;i++){
            totalsum += cardPoints[i];
        }
        int ans = totalsum;
        int idx = n-1;
        for(int i=k-1; i>=0;i--){
            totalsum -= cardPoints[i];
            totalsum += cardPoints[idx--];
            ans = max(ans, totalsum);
        }

        return ans;
    }
};
