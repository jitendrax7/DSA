class Solution {
public:
    vector<int> rowAndMaximumOnes(vector<vector<int>>& mat) {
        int n = mat.size() , m = mat[0].size();
        int idx = 0;
        int max_one = 0;
        for(int i= 0;i<n;i++){
            int count_one = 0;
            for(int j = 0;j<m;j++) if(mat[i][j]==1) count_one++;
            
            if(max_one<count_one){
                idx = i;
                max_one = count_one;
            }
        }

        return {idx,max_one};
    }
};
